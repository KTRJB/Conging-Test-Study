```swift
// 최대 최소 찾을 때 이진탐색 사용해보기

func 징검다리건너기(_ stones: [Int], _ k: Int) -> Int {
    // 이진 탐색의 왼쪽 경계(left)와 오른쪽 경계(right)를 설정
    var left = 1
    var right = 200000000

    // left와 right가 같아질 때까지 반복
    while left < right {
        var jump = 0 // 연속된 돌을 건너뛸 수 있는 개수를 추적
        let mid = (left + right) / 2 // 중간값(mid)을 계산하여 이진 탐색 진행

        // 모든 돌(stones)을 순회하면서 mid 값으로 계산
        for i in 0..<stones.count {
            // 만약 현재 돌(stones[i])이 mid보다 작은 값을 가지면, 건너뛸 수 없다고 판단
            if stones[i] - mid <= 0 {
                jump += 1 // 연속된 건너뛰는 돌의 개수를 증가시킴

                // 만약 jump가 k 이상이면 더 이상 돌을 건널 수 없으므로 중단
                if jump >= k {
                    break
                }
            } else {
                // 돌이 건너뛰기 불가능한 상태가 아니면 jump를 0으로 초기화
                jump = 0
            }
        }

        // 만약 jump가 k 이상이면, 중간 값이 너무 크므로 오른쪽 경계를 줄임
        if jump >= k {
            right = mid
        } else {
            // 그렇지 않으면 왼쪽 경계를 늘림
            left = mid + 1
        }
    }

    // 최종적으로 남은 left 값을 반환 (징검다리를 건널 수 있는 최소 하중)
    return left
}
```
