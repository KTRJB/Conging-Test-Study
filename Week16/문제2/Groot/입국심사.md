// gpt 코드 6점
func 입국심사(_ n: Int, _ times: [Int]) -> Int64 {
    var left: Int64 = 1
    var right: Int64 = Int64(times.max()! * n)
    var answer: Int64 = right

    while left <= right {
        let mid = (left + right) / 2
        var total = 0

        // 현재 주어진 시간(mid) 동안 각 심사관이 처리할 수 있는 사람 수를 계산
        for time in times {
            total += Int(mid / Int64(time))
        }

        if total >= n {
            // 처리할 수 있는 사람 수가 n명 이상이면 더 작은 시간으로 탐색
            answer = mid
            right = mid - 1
        } else {
            // n명을 처리할 수 없으면 더 큰 시간으로 탐색
            left = mid + 1
        }
    }

    return answer
}
