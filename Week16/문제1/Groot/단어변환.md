func 단어변환(_ begin:String, _ target:String, _ words:[String]) -> Int {
    // target이 words에 없으면 변환 불가능
      guard words.contains(target) else { return 0 }

      // 단어 집합 생성 (begin 포함)
      var wordSet = Set(words)
      wordSet.insert(begin)

      // BFS를 위한 큐
      var queue: [(String, Int)] = [(begin, 0)]
      var visited = Set<String>()

      while !queue.isEmpty {
          let (currentWord, depth) = queue.removeFirst()

          if currentWord == target {
              return depth
          }

          // 현재 단어에서 한 글자만 다른 모든 단어를 찾아 큐에 추가
          for i in 0..<currentWord.count {
              var chars = Array(currentWord)
              for char in "abcdefghijklmnopqrstuvwxyz" {
                  chars[i] = char
                  let newWord = String(chars)

                  if wordSet.contains(newWord) && !visited.contains(newWord) {
                      queue.append((newWord, depth + 1))
                      visited.insert(newWord)
                  }
              }
          }
      }

      // 변환 불가능
      return 0
}
