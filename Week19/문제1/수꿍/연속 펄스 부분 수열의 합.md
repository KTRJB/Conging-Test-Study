```swift
import Foundation

func programmers_week19_1(_ sequence: [Int]) -> Int64 {
    let n = sequence.count
    var pulseSequence1 = [Int](repeating: 0, count: n)
    var pulseSequence2 = [Int](repeating: 0, count: n)

    // sequence에 두 가지 형태의 펄스 수열을 적용하여 pulseSequence1, pulseSequence2 배열 생성
    for i in 0..<n {
        pulseSequence1[i] = i % 2 == 0 ? sequence[i] * 1 : sequence[i] * -1   // [1, -1, 1, -1, ...] 적용
        pulseSequence2[i] = i % 2 == 0 ? sequence[i] * -1 : sequence[i] * 1   // [-1, 1, -1, 1, ...] 적용
    }

    // 두 가지 펄스 수열에 대해 최대 부분합 계산
    let result1 = getMaxSum(pulseSequence1)
    let result2 = getMaxSum(pulseSequence2)

    // 두 결과 중 더 큰 값을 Int64로 변환하여 반환
    return Int64(max(result1, result2))
}

private func getMaxSum(_ arr: [Int]) -> Int {
    var maxSum = arr[0]     // 지금까지의 최대 부분합
    var currentSum = arr[0] // 현재까지의 연속된 부분합

    // 배열을 순회하며 최대 부분합 계산 (DP)
    for i in 1..<arr.count {
        currentSum = max(arr[i], currentSum + arr[i]) // 새로운 원소 포함 또는 새로 시작
        maxSum = max(maxSum, currentSum)              // 최대 부분합 갱신
    }

    return maxSum // 최대 부분합 반환
}
```
