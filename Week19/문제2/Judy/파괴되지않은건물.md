func solution(_ board:[[Int]], _ skill:[[Int]]) -> Int {
    var boards = board
    let boardC = board.first!.count
    let boardR = board.count
    var skills = Array(repeating: Array(repeating: 0, count: boardC + 1), count: boardR + 1)

    
    for sk in skill {
        let (ss, se, es, ee) = (sk[1], sk[2], sk[3], sk[4])
        let change = (sk[0] == 1) ? -sk[5] : sk[5]

        skills[ss][se] += change
        skills[es + 1][ee + 1] += change
        skills[es + 1][se] += -change
        skills[ss][ee + 1] += -change
    }
   
    for i in 0...boardR {
        for j in 1...boardC {
            skills[i][j] += skills[i][j - 1]
        }
    }
    
    for i in 1...boardR {
        for j in 0...boardC { 
            skills[i][j] += skills[i - 1][j]
        }
    }
    
    var counts = 0
    for i in 0..<boardR {
        for j in 0..<boardC {
            if boards[i][j] + skills[i][j] > 0 {
                counts += 1
            }
        }
    }
    
    return counts
}
