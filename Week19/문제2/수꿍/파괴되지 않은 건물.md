func programmers_week19_2(_ board: [[Int]], _ skill: [[Int]]) -> Int {
    var arr = board
    let n = arr.count
    let m = arr[0].count
    var diff = Array(repeating: Array(repeating: 0, count: m + 1), count: n + 1) // 차분 배열 diff를 원 배열보다 한 칸 더 큰 크기로 초기화

    for s in skill {
        let type = s[0]
        let degree = s[5] * (type == 1 ? -1 : 1)
        let r1 = s[1]
        let c1 = s[2]
        let r2 = s[3]
        let c2 = s[4]

        diff[r1][c1] += degree
        diff[r1][c2 + 1] -= degree
        diff[r2 + 1][c1] -= degree
        diff[r2 + 1][c2 + 1] += degree
    }

    // 차분 배열을 이용해 실제 배열로 변환
    for i in 0..<n {
        for j in 0..<m {
            if i > 0 { diff[i][j] += diff[i - 1][j] }
            if j > 0 { diff[i][j] += diff[i][j - 1] }
            if i > 0 && j > 0 { diff[i][j] -= diff[i - 1][j - 1] }
            arr[i][j] += diff[i][j]
        }
    }

    // 양수의 개수 세기
    let result = arr.flatMap { $0 }.filter { $0 > 0 }.count

    return result
}
