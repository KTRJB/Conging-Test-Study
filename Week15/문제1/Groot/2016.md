func 이공일육년(_ a: Int, _ b: Int) -> String {
    // 각 달의 일수를 저장한 배열 (2016년은 윤년이므로 2월은 29일)
    let numbers = [31, 29, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31]

    // 요일을 나타내는 배열 (2016년 1월 1일은 금요일이었으므로 금요일부터 시작)
    let days = ["FRI", "SAT", "SUN", "MON", "TUE", "WED", "THU"]

    // 날짜 계산을 위한 변수 초기화
    var result = 0

    // 주어진 달(a)까지의 총 일수를 계산
    numbers.enumerated().forEach {
        if $0.offset < a - 1 {
            // a월 이전의 모든 달의 일수를 더함
            result += $0.element
        } else if $0.offset == a - 1 {
            // 주어진 달(a)의 일(b)만큼 더함
            result += b - 1
        }
    }

    // 계산된 총 일수에 따라 요일을 반환
    return days[result % 7]
}
