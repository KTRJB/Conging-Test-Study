func 예상대진표(_ n:Int, _ a:Int, _ b:Int) -> Int
{
    // 라운드 수를 저장할 변수
        var answer = 0

        // a와 b의 현재 위치를 저장할 변수 (초기값으로 a와 b를 사용)
        var nextA = a
        var nextB = b

        // a와 b가 같은 라운드에 만날 때까지 반복
        repeat {
            // 각 참가자가 다음 라운드에 가면 번호가 절반이 됨 (홀수는 +1을 하고 2로 나눔)
            nextA = (nextA + 1) / 2
            nextB = (nextB + 1) / 2

            // 라운드 수 증가
            answer += 1
        } while nextA != nextB // a와 b가 같은 번호가 되면 종료

        // 두 참가자가 만나기까지의 라운드 수 반환
        return answer
}
