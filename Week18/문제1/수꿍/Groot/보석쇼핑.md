// 시간초과

func 보석쇼핑(_ gems:[String]) -> [Int] {
    let number = Set(gems).count

    guard number > 1 else {
        return [1, 1]
    }

    var head = 0
    var tail = 1
    var answer = [100000, 200000]


    while head < gems.count && tail < gems.count && head < tail {
        var temp = [String: Int]()

        for i in head...tail {
            let gem = gems[i]

            if temp[gem] == nil {
                temp[gem] = 1
            } else {
                temp[gem]! += 1
            }
        }

        if temp.count == number {
            if (answer[1] - answer[0]) > tail - head {
                answer[0] = head + 1
                answer[1] = tail + 1
            }

            if temp[gems[head]]! > 0 {
                temp[gems[head]]! -= 1
            } else {
                temp[gems[head]] = nil
            }

            head += 1
        } else {
            tail += 1
        }
    }

    return answer
}
